(defrule masinai-sviesoforas
	(masinos-vieta ?masinos-vieta ?masinos-pav)
	(sviesoforo-vieta ?masinos-vieta ?sviesoforo-pav)

=>
	(printout t "masina " ?masinos-pav " yra " ?masinos-vieta " gatveje ir yra prie " ?sviesoforo-pav " sviesoforo" crlf)
	(assert (masina-prie-sviesoforo ?masinos-pav ?sviesoforo-pav))
	(assert (masinai-spalva ?masinos-pav ?sviesoforo-pav))

)


(defrule masinai-spalva
	(masina-prie-sviesoforo ?masinos-pav ?sviesoforo-pav)

	(printout t "masinai " ?masinos-pav " dega " ?spalva " spalva" crlf)
